Splunk SPL queries for detection
1. Bulk USB copy — volume anomaly (>500 MB in session)
index=endpoint sourcetype=wineventlog EventCode=4663 Object_Type="File"
| eval mb = round(size_bytes / 1048576, 2)
| stats sum(mb) as total_mb by user, drive_letter, _time
| where total_mb > 500 AND match(drive_letter, "[D-Z]:")
| eval severity="CRITICAL"
| table _time, user, drive_letter, total_mb, severity
2. Off-hours file access scoring
index=endpoint sourcetype=wineventlog EventCode IN (4663,4656)
| eval hour = strftime(_time, "%H")
| where (hour >= "00" AND hour <= "06") OR hour >= "22"
| stats count as offhours_events by user
| where offhours_events > 5
| eval risk_score = offhours_events * 3
| sort -risk_score
3. Email attachment exfiltration (DLP)
index=email_gateway sourcetype=exchange_dlp
| where isnotnull(attachment_size_kb) AND dest_domain != "corp.local"
| eval mb = round(attachment_size_kb/1024,1)
| stats sum(mb) as total_mb, dc(dest_email) as uniq_dest by user
| where total_mb > 50 OR uniq_dest > 10
| table user, total_mb, uniq_dest
4. Composite insider threat risk score
| multisearch
  [ search index=endpoint EventCode=4663 | stats count as file_events by user ]
  [ search index=dlp | stats count as dlp_events by user ]
  [ search index=email offhours=true | stats count as offhours by user ]
| stats sum(file_events) as fe, sum(dlp_events) as de, sum(offhours) as oh by user
| eval risk = (fe*0.3) + (de*5) + (oh*3)
| sort -risk | head 10
5. Log tampering / event log cleared
index=wineventlog (EventCode=1102 OR EventCode=104)
| eval action=if(EventCode=1102, "Security log cleared", "System log cleared")
| table _time, host, user, action
| eval severity="CRITICAL"
| sendalert "SOC-ALERT" param.urgency=critical
